# 志愿填报助手 - 详细实施方案

## 一、技术架构决策

### 1.1 架构选型：Monorepo 单体优先

**选择理由**：
- 初期团队小，微服务增加运维复杂度
- 代码共享方便，类型定义统一
- 后期可平滑拆分为微服务

```
架构演进路线：
Phase 1: Monorepo 单体应用 (MVP)
Phase 2: 模块化单体 (功能完善期)
Phase 3: 按需拆分微服务 (规模扩张期)
```

### 1.2 技术栈最终选型

| 层级 | 技术 | 版本 | 选择理由 |
|------|------|------|----------|
| **前端框架** | Next.js | 14.x | App Router、SSR/SSG、SEO友好 |
| **UI组件库** | Ant Design | 5.x | 企业级组件、中文友好 |
| **状态管理** | Zustand | 4.x | 轻量、TypeScript友好 |
| **数据请求** | TanStack Query | 5.x | 缓存、乐观更新、离线支持 |
| **图表库** | ECharts | 5.x | 功能强大、中文文档完善 |
| **后端框架** | NestJS | 10.x | 模块化、依赖注入、企业级 |
| **ORM** | Prisma | 5.x | 类型安全、迁移管理、性能好 |
| **数据库** | PostgreSQL | 16.x | 稳定、JSON支持、全文搜索 |
| **缓存** | Redis | 7.x | 高性能、数据结构丰富 |
| **AI接口** | Claude API | - | 智能推荐、对话助手 |
| **认证** | JWT + Redis | - | 无状态、可撤销 |
| **文件存储** | MinIO/阿里OSS | - | S3兼容、成本低 |

```prisma
// ==================== 院校相关 ====================

model University {
  id        Int      @id @default(autoincrement())
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // 基本信息
  name     String  @db.VarChar(200)
  code     String? @unique @db.VarChar(50)
  province String? @db.VarChar(50)
  city     String? @db.VarChar(100)

  // 院校属性
  type          String? @map("university_type") @db.VarChar(50)      // 综合/理工/师范等
  level         String? @map("university_level") @db.VarChar(50)     // 本科/专科
  runningLevel  String? @map("running_level") @db.VarChar(50)        // 办学等级
  runningNature String? @map("running_nature") @db.VarChar(50)       // 公办/民办

  // 院校标签
  isDoubleFirstClass Boolean  @default(false) @map("is_double_first_class")
  is985              Boolean  @default(false) @map("is_985")
  is211              Boolean  @default(false) @map("is_211")
  tags               String[] @map("university_tags")
  grade              String?  @map("university_grade") @db.VarChar(50)

  // 学科建设
  disciplineEvaluationLevel String?  @map("discipline_evaluation_level") @db.VarChar(50)
  hasMasterProgram          Boolean  @default(false) @map("has_master_program")
  hasDoctoralProgram        Boolean  @default(false) @map("has_doctoral_program")
  masterProgramCount        Int?     @map("master_program_count")
  masterPrograms            String[] @map("master_programs")
  doctoralProgramCount      Int?     @map("doctoral_program_count")
  doctoralPrograms          String[] @map("doctoral_programs")

  // 其他属性
  campusArea             Decimal? @map("campus_area") @db.Decimal(10, 2)
  transferDifficulty     String?  @map("transfer_difficulty") @db.VarChar(50)
  militaryTrainingDuration String? @map("military_training_duration") @db.VarChar(50)
  isFeatured             Boolean  @default(false) @map("is_featured")

  // 备注
  notes String? @db.Text

  // 关联
  enrollmentPlans  EnrollmentPlan[]
  admissionRecords AdmissionRecord[]
  favorites        Favorite[]

  @@index([province])
  @@index([city])
  @@index([type])
  @@index([isDoubleFirstClass])
  @@index([is985])
  @@index([is211])
  @@map("universities")
}

model Major {
  id        Int      @id @default(autoincrement())
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // 基本信息
  name String  @db.VarChar(200)
  code String? @db.VarChar(50)

  // 专业分类
  category   String? @map("major_category") @db.VarChar(100)   // 专业大类
  level      String? @map("major_level") @db.VarChar(50)       // 本科/专科
  discipline String? @db.VarChar(100)                          // 学科门类
  type       String? @map("major_type") @db.VarChar(50)

  // 专业属性
  notes        String?  @map("major_notes") @db.Text
  isRestricted Boolean  @default(false) @map("is_restricted")

  // 就业信息
  employmentRate     Decimal? @map("employment_rate") @db.Decimal(5, 2)
  avgSalary          Int?     @map("avg_salary")
  employmentDirection String? @map("employment_direction") @db.Text

  // 关联
  enrollmentPlans  EnrollmentPlan[]
  admissionRecords AdmissionRecord[]
  favorites        Favorite[]

  @@index([category])
  @@index([level])
  @@index([discipline])
  @@map("majors")
}
```

```typescript
// 推荐算法核心逻辑
class RecommendationEngine {
  /**
   * 冲稳保策略计算
   *
   * 核心原理：
   * - 冲：用户位次 + rushRange（如+10000）范围��的院校专业
   * - 稳：用户位次 ± stableRange（如±3000）范围内的院校专业
   * - 保：用户位次 - safeRange（如-5000~-10000）范围内的院校专业
   */

  // 位次差计算
  calculateRankDiff(userRank: number, admissionRank: number): number {
    return admissionRank - userRank;
  }

  // 录取概率计算（基于历史数据和位次差）
  calculateAcceptRate(rankDiff: number, historicalData: AdmissionRecord[]): number {
    // 基础概率模型
    if (rankDiff > 10000) return 0.1;      // 冲刺，10%
    if (rankDiff > 5000) return 0.3;       // 较冲，30%
    if (rankDiff > 0) return 0.5;          // 略冲，50%
    if (rankDiff > -3000) return 0.7;      // 稳妥，70%
    if (rankDiff > -5000) return 0.85;     // 较稳，85%
    return 0.95;                            // 保底，95%
  }

  // 策略分类
  classifyStrategy(rankDiff: number): 'rush' | 'stable' | 'safe' {
    if (rankDiff > 3000) return 'rush';
    if (rankDiff > -3000) return 'stable';
    return 'safe';
  }

  // 综合评分（考虑多维度因素）
  calculateScore(item: RecommendCandidate, preferences: UserPreferences): number {
    let score = 0;

    // 1. 录取概率权重 (40%)
    score += item.acceptRate * 40;

    // 2. 院校质量权重 (25%)
    if (item.university.is985) score += 25;
    else if (item.university.is211) score += 20;
    else if (item.university.isDoubleFirstClass) score += 15;
    else score += 10;

    // 3. 地域偏好权重 (15%)
    if (preferences.provinces?.includes(item.university.province)) {
      score += 15;
    }

    // 4. 专业匹配权重 (20%)
    if (preferences.majorCategories?.includes(item.major.category)) {
      score += 20;
    }

    return score;
  }
}
```

```typescript
export class DataImporter {
  private prisma: PrismaClient;
  private universityCache: Map<string, number> = new Map();
  private majorCache: Map<string, number> = new Map();

  async importFromExcel(filePath: string, province: string) {
    const workbook = XLSX.readFile(filePath);
    const sheet = workbook.Sheets[province];
    const rows: ExcelRow[] = XLSX.utils.sheet_to_json(sheet);

    console.log(`开始导入 ${rows.length} 条数据...`);

    // 分批处理，每批 100 条
    const batchSize = 100;
    for (let i = 0; i < rows.length; i += batchSize) {
      const batch = rows.slice(i, i + batchSize);
      await this.processBatch(batch, province);
      console.log(`已处理 ${Math.min(i + batchSize, rows.length)}/${rows.length}`);
    }
  }

  private async processBatch(rows: ExcelRow[], province: string) {
    for (const row of rows) {
      // 1. 处理院校（去重）
      const universityId = await this.upsertUniversity(row);

      // 2. 处理专业（去重）
      const majorId = await this.upsertMajor(row);

      // 3. 处理招生计划
      await this.upsertEnrollmentPlan(row, universityId, majorId, province);

      // 4. 处理录取数据（多年）
      await this.upsertAdmissionRecords(row, universityId, majorId, province);
    }
  }

  private async upsertUniversity(row: ExcelRow): Promise<number> {
    const code = row['院校代码'];

    // 检查缓存
    if (this.universityCache.has(code)) {
      return this.universityCache.get(code)!;
    }

    const university = await this.prisma.university.upsert({
      where: { code },
      update: {},
      create: {
        name: row['院校'],
        code: code,
        province: row['院校省份'],
        city: row['院校城市'],
        type: row['院校类型'],
        level: row['院校层次'],
        runningNature: row['办学性质'],
        isDoubleFirstClass: row['是否双一流'] === '是',
        tags: row['院校标签']?.split(',') || [],
      },
    });

    this.universityCache.set(code, university.id);
    return university.id;
  }

  // ... 其他方法类似
```

```typescript
// 权限守卫
@Injectable()
export class RolesGuard implements CanActivate {
  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.get<UserRole[]>('roles', context.getHandler());
    if (!requiredRoles) return true;

    const request = context.switchToHttp().getRequest();
    const user = request.user;

    return requiredRoles.includes(user.role);
  }
}

// VIP 功能守卫
@Injectable()
export class VipGuard implements CanActivate {
  canActivate(context: ExecutionContext): boolean {
    const requiredLevel = this.reflector.get<VipLevel>('vipLevel', context.getHandler());
    if (!requiredLevel) return true;

    const request = context.switchToHttp().getRequest();
    const user = request.user;

    const levelOrder = { FREE: 0, BASIC: 1, PREMIUM: 2, EXPERT: 3 };
    return levelOrder[user.vipLevel] >= levelOrder[requiredLevel];
  }
}
```
